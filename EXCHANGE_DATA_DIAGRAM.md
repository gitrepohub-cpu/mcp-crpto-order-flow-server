## üíæ In-Memory Data Storage Schema

```python
# Price Data
prices[symbol][exchange] = {
    "price": float,          # Mid price
    "bid": float,            # Best bid
    "ask": float,            # Best ask
    "timestamp": int,        # Unix milliseconds
    "exchange": str          # Exchange name
}

# Mark/Index Prices (Futures only)
mark_prices[symbol][exchange] = {
    "mark_price": float,     # Mark/fair price
    "index_price": float,    # Spot index price
    "basis": float,          # mark - index
    "basis_pct": float       # Basis as percentage
}

# Funding Rates (Futures only)
funding_rates[symbol][exchange] = {
    "funding_rate": float,   # Rate (0.0001 = 0.01%)
    "rate_pct": float,       # As percentage
    "annualized_rate": float,# APR %
    "next_funding_time": int # Unix ms
}

# Open Interest (Futures only)
open_interest[symbol][exchange] = {
    "open_interest": float,       # Number of contracts
    "open_interest_value": float, # USD value
    "timestamp": int
}

# 24H Ticker Statistics
ticker_24h[symbol][exchange] = {
    "volume_24h": float,              # Base asset volume
    "quote_volume_24h": float,        # Quote (USD) volume
    "high_24h": float,                # 24h high price
    "low_24h": float,                 # 24h low price
    "price_change_percent_24h": float # 24h change %
}

# Orderbook Depth
orderbooks[symbol][exchange] = {
    "bids": [[price, qty], ...],  # Sorted high to low
    "asks": [[price, qty], ...],  # Sorted low to high
    "timestamp": int
}

# Recent Trades (Ring buffer, max 100)
trades[symbol][exchange] = [
    {"price": float, "quantity": float, "side": str, "timestamp": int},
    ...
]

# Liquidations (Ring buffer, max 50)
liquidations[symbol] = [
    {"exchange": str, "side": str, "price": float, "quantity": float, "timestamp": int},
    ...
]

# Candles (Latest 1-minute)
candles[symbol][exchange] = {
    "open": float, "high": float, "low": float, "close": float,
    "volume": float, "open_time": int, "close_time": int
}
```
